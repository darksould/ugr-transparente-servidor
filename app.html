Dependencias

  • Express: Framework web
var express = require('express'); var app = express();

Dependencias Locales

  • Server Config: Configuración del servidor
var serverConfig = require('./app/serverConfig');
  • Routes: Rutas del servidor
var routes = require('./app/routes'); serverConfig(app); routes(app);

Se levanta la aplicación en el puerto e ip determinados

var server = app.listen(app.get('port'), app.get('ip'), function() { console.log("UGR - Transparente " + process.env.npm_package_version); console.log('Express server listening on ' + app.get('ip') + ':' + app.get('port')); });
